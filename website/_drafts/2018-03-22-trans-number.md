### parseInt()

```js
parseInt('123') // -> 123
parseInt('123s') // -> 123
parseInt('s123') // -> NaN
parseInt('123.3') // -> 123
parseInt('123.3s') // -> 123
parseInt('s123.3') // -> NaN
```

### parseFloat()

```js
parseFloat('123.3') // -> 123.3
parseFloat('123.3s') // -> 123.3
parseFloat('s123.3') // -> NaN
```

### Number()

```js
Number('123') // -> 123
Number('123s') // -> NaN
Number('s123') // -> NaN
Number('123.3') // -> 123.3
Number('123.3s') // -> NaN
Number('s123.3') // -> NaN
```

### +

```js
+'123' // -> 123
+'123s' // -> NaN
+'s123' // -> NaN
+'123.3' // -> 123.3
+'123.3s' // -> NaN
+'s123.3' // -> NaN
```

### >>

```js
'123'>>0 // -> 123
'123s'>>0 // -> 0
's123'>>0 // -> 0
'123.3'>>0 // -> 123
'123.3s'>>0 // -> 0
's123.3'>>0 // -> 0
```

### ~~

```js
~~'123' // -> 123
~~'123s' // -> 0
~~'s123' // -> 0
~~'123.3' // -> 123
~~'123.3s' // -> 0
~~'s123.3' // -> 0
```

### |

```js

'123s'|0 // -> 0
's123'|0 // -> 0
'123.3'|0 // -> 123
'123.3s'|0 // -> 0
's123.3'|0 // -> 0
```
